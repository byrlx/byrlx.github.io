ELISP>

recursive function 

ELISP> (defun in_list (obj lst)
	 (if (null lst)
	     nil
	   (if (eql obj (car lst))
	       lst
	     (in_list obj (cdr lst)))))
in_list
ELISP> (in_list 'y '(a b c x y z))
(y z)
