-------------------
		
	set nu
	set cindent 
	set autoindent
	set smartindent
	
	# ask if save unsaved files
	set confirm
	
	set tabstop=4
	set shiftwidth=4
	set noexpandtab
	set smarttab
	
	syntax on
	syntax enable
	
	filetype indent on
	
	# high light the search result 
	set hlsearch
	set is

	# command completion
	set wildmenu
	
	#  turn off vi compatibility
	set nocp

	# highlight current line
	set cul
	
	set encoding=utf-8
	set fileencoding=utf-8
	set termencoding=utf-8
	
	# tlist related option
	let Tlist_Auto_Open = 1
	let Tlist_Exit_OnlyWindow = 1
	let Tlist_WinWidth = 30
